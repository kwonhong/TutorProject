CREATE DATABASE IF NOT EXISTS footyfixdatabase;

USE footyfixdatabase;
CREATE TABLE IF NOT EXISTS userdata (
id INT auto_increment,
username TEXT NOT NULL,
password TEXT NOT NULL,
firstname TEXT,
lastname TEXT,
address TEXT,
city TEXT,
country TEXT,
email TEXT,
postalcode TEXT,
age int,
gender TEXT,
geolat DOUBLE,
geolon DOUBLE,
PRIMARY KEY (id)
);

CREATE TABLE IF NOT EXISTS event (
id INT auto_increment,
name TEXT,
address TEXT,
city TEXT,
country TEXT,
postalcode TEXT,
type TEXT,
gender TEXT,
date date,
minimumAge int,
maximumAge int,
numParticipants int DEFAULT 0,
capacity int,
description TEXT,
geolat DOUBLE,
geolon DOUBLE,
PRIMARY KEY (id)
);
