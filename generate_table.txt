CREATE TABLE IF NOT EXISTS user (
ID TEXT,
Password TEXT,
FirstName TEXT,
LastName TEXT,
Address TEXT,
City TEXT,
Email TEXT,
PostalCode TEXT,
Age int,
Gender TEXT
);

CREATE TABLE IF NOT EXISTS event (
ID TEXT,
Name TEXT,
Address TEXT,
City TEXT,
PostalCode TEXT,
Type TEXT,
Gender TEXT,
Date date,
startTime time,
minimumAge int,
maximumAge int,
numParticipants int,
Capacity int,
Description TEXT
);
